{"name": "echo","id": 1},{"name": "print","id": 2},{"name": "hello world","id": 3}

def  test_search():  
  with open(“test_data.json”) as data_file: 
    data_text = json.load(data_file) 
      for each in data_text: 
        search_text = each["name"] 
        var = driver.find_element_by_id("search_box")
        var.clear()
        var.send_keys("search_text") 
        driver.find_element_by_name("search").click() 
        element = driver.find_elements_by_class_name('hl') 
        element_text = {z.text.lower() for z in element}
        self.assertIn(search_text.lower(), element_text, "Search result does not match")

Behaviour Driven Testing

Sample Feature:


#search.feature						
Feature: To allow the users to search for free text
  Scenario: Search for keyword 'echo'
  Given that customer is on the home page
  When he searches for 'echo'
  Then he should see keyword 'echo' highlighted in the search results

Sample test script:

./features/step_definitions/search.rb:


Given(/^that customer am on the home page$/) do
  @browser.goto(@url)
end

When(/^customer search for 'echo'$/) do
  @browser.text_field(:name => "q").set "echo"
 @browser.text_field(:name => "q").send_keys :return
end

Then(/^customer should see keyword 'echo' highlighted in the search results$/) do
  @browser.spans(:class => "hl").each do |query|
    query.text.should == "echo"
  end
end
